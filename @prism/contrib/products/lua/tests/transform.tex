\directlua{
  PALNAME = "\PALETTE"
  
  PALSIZE = 10

  PALDIM   = 1
  PALDELTA = .3

  xmin, xmax = -100, 100

  COL_1 = "lightgray!95"
  COL_2 = "lightgray!20!black"
}


\begin{luacode*}
function drawpal(
  palname,
  palsize,
  options
)
palette = getPal(palname, options)

local g = graph:new{
  window = {xmin, xmax, -5, 5},
  bbox   = false
}

g:Linewidth(10)

local A = Z(-20, 4)
local v = Z(0, -PALDIM)

for k = 1, palsize do
  local color = rgb(palette[k])

  g:Drectangle(
    A, A + PALDIM, A + PALDIM + v,
    "color = " .. COL_1 .. ", fill = " .. color
  )

  g:Dlabel(
    "\\Large$" .. k .. "$",
    A + Z(PALDIM, 1.15*PALDIM) / 2,
    {pos="S"}
  )

  A = A + PALDIM + PALDELTA
end

g:Show()

end
\end{luacode*}
