  PALDELTA = .3

  COL_2 = "lightgray!20!black"
}


\begin{luacode*}
function drawpal(
  palette,
  palsize,
  options
)
new_palette = palCreateFromPal(palette, options)

local g = graph:new{
  window = {xmin, xmax, -5, 5},
  bbox   = false
}

g:Linewidth(10)

local A = Z(-20, 4)
local v = Z(0, -PALDIM)

for k = 1, palsize do
  local color = rgb(new_palette[k])

  g:Drectangle(
    A, A + PALDIM, A + PALDIM + v,
    "color = " .. COL_1 .. ", fill = " .. color
  )

  g:Dlabel(
    "\\Large$" .. k .. "$",
    A + Z(PALDIM, 1.15*PALDIM) / 2,
    {pos="S"}
  )

  A = A + PALDIM + PALDELTA
end

g:Show()

end
\end{luacode*}
