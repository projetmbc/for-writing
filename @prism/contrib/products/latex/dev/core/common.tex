\ExplSyntaxOn

% Variables used.
\ior_new:N   \g_dev_file_ior
\ior_open:Nn \g_dev_file_ior { pal-def.sty }

\iow_new:N   \g_final_file_iow
\iow_open:Nn \g_final_file_iow { PALETTE-PROJECT.sty }

\bool_new:N       \g_in_header_bool
\bool_set_false:N \g_in_header_bool

\seq_new:N   \g_magic_comments_seq
\seq_clear:N \g_magic_comments_seq

\seq_new:N   \g_tex_code_seq
\seq_clear:N \g_tex_code_seq

\seq_new:N \g_palette_PALETTE_BEGIN_seq
\tl_new:N  \g_palette_last_col_tl


% Lines from the user's def.
\ior_str_map_inline:Nn \g_dev_file_ior {
  \regex_if_match:nnTF { \%{3}\s* }{ #1 }{
% Line == '%%%'.
    \bool_set_inverse:N \g_in_header_bool

% Line != '%%%'.
  }{
% In ''this'' header.
    \bool_if:NTF \g_in_header_bool {
      \seq_put_right:Nn \g_magic_comments_seq { #1}
% Line of code?
    }{
      \seq_put_right:Nn \g_tex_code_seq { #1}
    }
  }
}


% Trim ''\g_tex_code_seq''.
\bool_while_do:nn {
  ! \seq_if_empty_p:N \g_tex_code_seq &&
  \tl_if_blank_p:e { \seq_item:Nn \g_tex_code_seq { 1 } }
}{
  \seq_pop_left:NN \g_tex_code_seq \l_tmpa_tl
}

\bool_while_do:nn {
  ! \seq_if_empty_p:N \g_tex_code_seq &&
  \tl_if_blank_p:e { \seq_item:Nn \g_tex_code_seq { -1 } }
}
{ \seq_pop_right:NN \g_tex_code_seq \l_tmpa_tl }


% Write into the final file.
\iow_now:Nx \g_final_file_iow {
  \c_percent_str
  \c_percent_str
  \c_percent_str
}

\seq_map_inline:Nn \g_magic_comments_seq {
  \iow_now:Nx \g_final_file_iow { #1 }
}

\seq_if_empty:NF \g_magic_comments_seq {
  \iow_now:Nx \g_final_file_iow {
    \c_percent_str
  }

  \iow_now:Nx \g_final_file_iow {
    \c_percent_str
  }
}

\iow_now:Nx \g_final_file_iow {
  \c_percent_str
  \c_space_tl
  Here \c_space_tl is    \c_space_tl
  the  \c_space_tl LaTeX \c_space_tl
  code \c_space_tl used.
}

\iow_now:Nx \g_final_file_iow {
  \c_percent_str
}

\iow_now:Nx \g_final_file_iow {
  \c_percent_str
  \c_space_tl
  latex::
}

\seq_map_inline:Nn \g_tex_code_seq {
  \iow_now:Nx \g_final_file_iow {
    \c_percent_str
    \c_space_tl
    \c_space_tl \c_space_tl
    \c_space_tl \c_space_tl
    #1
  }
}

\iow_now:Nx \g_final_file_iow {
  \c_percent_str
  \c_percent_str
  \c_percent_str
}

\iow_now:Nx \g_final_file_iow {}

\iow_now:Ne \g_final_file_iow {
  \c_backslash_str palCreateFromRGB
  \c_left_brace_str PALETTE \c_right_brace_str
  \c_left_brace_str
}

% Color definitions without unseful last coma.
\seq_set_eq:NN
  \g_palette_PALETTE_BEGIN_seq
  \g_palette_PALETTE_seq

\seq_pop_right:NN
  \g_palette_PALETTE_BEGIN_seq
  \g_palette_last_col_tl

\seq_map_inline:Nn \g_palette_PALETTE_BEGIN_seq {
  \iow_now:Nx \g_final_file_iow {
    \c_space_tl \c_space_tl
    { #1 },
  }
}

\iow_now:Nx \g_final_file_iow {
  \c_space_tl \c_space_tl
  \c_left_brace_str \g_palette_last_col_tl \c_right_brace_str
}

\iow_now:Ne \g_final_file_iow {
  \c_right_brace_str
}

\iow_close:N \g_final_file_iow
\ior_close:N \g_dev_file_ior

\ExplSyntaxOff
