\begin{tikzpicture}
  \pgfmathsetmacro{\largeur}{12}
  \pgfmathsetmacro{\hauteur}{1.5}

  % Création du spectre continu
  \foreach \i in {1,...,\inteval{\nbcol-1}} {
    \pgfmathsetmacro{\xstart}{\largeur*(\i-1)/(\nbcol-1)}
    \pgfmathsetmacro{\xend}{\largeur*\i/(\nbcol-1)}
    \shade[left color=\palUse{PALETTE}{\i}, right color=\palUse{PALETTE}{\inteval{\i+1}}]
      (\xstart, 0) rectangle (\xend, \hauteur);
  }

  % Bordure
  \draw[COL1, thick] (0, 0) rectangle (\largeur, \hauteur);
\end{tikzpicture}
