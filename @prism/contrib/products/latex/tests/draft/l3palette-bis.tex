\documentclass{article}
\usepackage{xcolor}
\usepackage{tikz}
\usepackage{expl3}

\ExplSyntaxOn
% Définition des palettes
\definecolor{palette@Blackbody@1}{rgb}{0.0,0.0,0.0}
\definecolor{palette@Blackbody@2}{rgb}{0.2,0.0,0.0}
\definecolor{palette@Blackbody@3}{rgb}{0.4,0.0,0.0}
\definecolor{palette@Blackbody@4}{rgb}{0.6,0.2,0.0}
\definecolor{palette@Blackbody@5}{rgb}{0.8,0.4,0.0}
\definecolor{palette@Blackbody@6}{rgb}{1.0,0.8,0.0}

\definecolor{palette@Ocean@1}{rgb}{0.0,0.2,0.4}
\definecolor{palette@Ocean@2}{rgb}{0.0,0.4,0.6}
\definecolor{palette@Ocean@3}{rgb}{0.0,0.6,0.8}
\definecolor{palette@Ocean@4}{rgb}{0.2,0.8,0.9}
\definecolor{palette@Ocean@5}{rgb}{0.4,1.0,1.0}

% Commande pour obtenir le nom de couleur "valide"
\cs_new:Npn \paletteColorName #1#2 { palette@#1@#2 }
\ExplSyntaxOff

\begin{document}

\begin{tikzpicture}
  % Blackbody
  \foreach \i in {1,...,6} {
    \fill [color=\csname palette@Blackbody@\i\endcsname] (\i,0) rectangle ++(1,1);
  }

  % Ocean
  \foreach \i in {1,...,5} {
    \fill [color=\csname palette@Ocean@\i\endcsname] (\i,2) rectangle ++(1,1);
  }
\end{tikzpicture}

\end{document}
