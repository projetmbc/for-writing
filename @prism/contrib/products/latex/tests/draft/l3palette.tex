\documentclass{article}
\usepackage{xcolor}
\usepackage{expl3}
\usepackage{tikz}

\ExplSyntaxOn

% =================================================
% PALETTES — STOCKAGE
% =================================================

\seq_new:N \l__palette_work_seq

\cs_new_protected:Nn \palette_new:n
  { \seq_new:c { g_palette_#1_seq } }

\cs_new_protected:Nn \palette_add_color:nn
  {
    \seq_gput_right:cn { g_palette_#1_seq } { #2 }
    \int_set:Nn \l_tmpa_int { \seq_count:c { g_palette_#1_seq } }
    \definecolor{palette@#1@\int_use:N\l_tmpa_int}{rgb}{#2}
  }

\cs_new_protected:Nn \palette_add_colors:nn
  {
    \clist_map_inline:nn { #2 }
      { \palette_add_color:nn { #1 } { ##1 } }
  }

\cs_new_protected:Nn \palette_create:nn
  {
    \palette_new:n { #1 }
    \palette_add_colors:nn { #1 } { #2 }
  }

\cs_new:Nn \palette_get_color_name:nn
  { palette@#1@#2 }

% =================================================
% OPÉRATIONS SUR LA PALETTE COURANTE
% =================================================

\cs_new_protected:Nn \palette_from:n
  { \seq_set_eq:NN \l__palette_work_seq g_palette_#1_seq }

\cs_new_protected:Nn \palette_reverse:
  { \seq_reverse:N \l__palette_work_seq }

\cs_new_protected:Nn \palette_select:n
  {
    \seq_clear:N \l_tmpa_seq
    \clist_map_inline:nn { #1 }
      {
        \seq_put_right:Nn \l_tmpa_seq
          { \seq_item:Nn \l__palette_work_seq { ##1 } }
      }
    \seq_set_eq:NN \l__palette_work_seq \l_tmpa_seq
  }

\cs_new_protected:Nn \palette_shift:n
  {
    \int_set:Nn \l_tmpa_int { #1 }
    \int_mod:nn { \l_tmpa_int } { \seq_count:N \l__palette_work_seq }
    \int_compare:nNnTF { \l_tmpa_int } < { 0 }
      { \int_add:Nn \l_tmpa_int { \seq_count:N \l__palette_work_seq } }
      { }

    \int_step_inline:nn { \l_tmpa_int }
      {
        \seq_pop_right:NN \l__palette_work_seq \l_tmpb_tl
        \seq_put_left:NV \l__palette_work_seq \l_tmpb_tl
      }
  }

\cs_new_protected:Nn \palette_to:n
  {
    \palette_new:n { #1 }
    \seq_map_inline:Nn \l__palette_work_seq
      { \palette_add_color:nn { #1 } { ##1 } }
  }

% =================================================
% INTERFACE UTILISATEUR
% =================================================

\NewDocumentCommand \palCreate { m m }
  { \palette_create:nn { #1 } { #2 } }

\NewDocumentCommand \paletteFrom { m }
  { \palette_from:n { #1 } }

\NewDocumentCommand \paletteReverse { }
  { \palette_reverse: }

\NewDocumentCommand \paletteSelect { m }
  { \palette_select:n { #1 } }

\NewDocumentCommand \paletteShift { m }
  { \palette_shift:n { #1 } }

\NewDocumentCommand \paletteTo { m }
  { \palette_to:n { #1 } }

% ⚠️ CORRECTION CRITIQUE ICI
\NewExpandableDocumentCommand \palUse { m m }
  { color=\palette_get_color_name:nn { #1 } { #2 } }

\ExplSyntaxOff

% =================================================
% EXEMPLE
% =================================================

\begin{document}

\palCreate{Blackbody}{
 {0,0,0},
 {0.4,0,0.2},
 {0.8,0.2,0},
 {1,0.6,0},
 {1,1,0.4},
 {1,1,1}
}

\paletteFrom{Blackbody}
  \paletteReverse
  \paletteSelect{2,3,4}
  \paletteShift{1}
\paletteTo{BlackbodyCustom}

\begin{tikzpicture}
\foreach \i in {1,...,3}
  \fill[\palUse{BlackbodyCustom}{\i}] (\i,0) rectangle ++(1,1);
\end{tikzpicture}

\end{document}
