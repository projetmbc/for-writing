% ------------------ %
% -- COLORED TEXT -- %
% ------------------ %

\bigskip\bigskip

\ExplSyntaxOn

\tl_new:N  \l_colored_text_tl
\tl_set:Nn \l_colored_text_tl {Some Colored Text...}

\tl_replace_all:Nnn \l_colored_text_tl { ~ } { \space }

\int_new:N \l_my_index_int
\int_zero:N \l_my_index_int

{\Large\bfseries
  \tl_map_inline:Nn \l_colored_text_tl {
    \int_incr:N \l_my_index_int

    \int_set:Nn \l_tmpa_int
        { \int_mod:nn { \l_my_index_int - 1 } { \nbcol } + 1 }

    \textcolor{\palUse{\PALETTE}{\int_use:N \l_tmpa_int}}{#1}
  }
}

\ExplSyntaxOff


% ----------------------- %
% -- CIRCULAR GRADIENT -- %
% ----------------------- %

\bigskip\bigskip

\begin{tikzpicture}
  \foreach \i in {1,...,\nbcol} {
    \pgfmathsetmacro{\rayon}{2.5 - 2.5*(\i-1)/\nbcol}
    \fill[\palUse{\PALETTE}{\i}] (0, 0) circle (\rayon);
  }
  % Bordure extérieure
  \draw[COL1, thick] (0, 0) circle (2.5);
\end{tikzpicture}


\ifthenelse{\nbcol < 41}{

  % ------------------------ %
  % -- WAVES (small size) -- %
  % ------------------------ %

  \bigskip\bigskip

  \def\waveWidth{16}
  \def\amplitude{.75}

  \begin{tikzpicture}
    \foreach \i [evaluate=\i as \yBase using (\i * 0.2)] in {1,...,\nbcol} {
      \draw[
        COL1, % Ou votre couleur dynamique
        draw = \palUse{\PALETTE}{\i},
        thick,
        line width = 2pt,
        line cap = round,
        fill = none
      ]
      plot[
        domain=0:\waveWidth,
        samples=100,
        smooth
      ]
      (\x, {\yBase + \amplitude * sin((\x * 4 + \i) r)});
    }
  \end{tikzpicture}
}{}


% -------------- %
% -- SPECTRUM -- %
% -------------- %

\bigskip\bigskip

\pgfmathsetmacro{\palspectwidth}{18}
\pgfmathsetmacro{\palspectheight}{1}

\begin{tikzpicture}
  \foreach \i in {1,...,\inteval{\nbcol-1}} {
    \pgfmathsetmacro{\xstart}{\palspectwidth*(\i-1)/(\nbcol-1)}
    \pgfmathsetmacro{\xend}{\palspectwidth*\i/(\nbcol-1)}
    \shade[left color  = \palUse{\PALETTE}{\i},
           right color = \palUse{\PALETTE}{\inteval{\i+1}}]
      (\xstart, 0) rectangle (\xend, \palspectheight);
  }

  \draw[COL1, thick]
    (0, 0) rectangle (\palspectwidth, \palspectheight);
\end{tikzpicture}


% ------------------------------- %
% -- PALETTE (small size only) -- %
% ------------------------------- %

\ifthenelse{\nbcol < 41}{
  \bigskip

  \begin{tikzpicture}
    \pgfmathsetmacro{\w}{\palspectwidth / \nbcol}

    \foreach \i in {1,...,\nbcol} {
        \pgfmathsetmacro{\xpos}{(\i-1) * \w}

        \draw[COL1, thick, fill = \palUse{\PALETTE}{\i}]
          (\xpos, 0) rectangle (\xpos + \w, \palspectheight);
    }
  \end{tikzpicture}
}{}
