def log_completion(message: str):
    logger = logging.getLogger()

    file_handler = next(
        h
        for h in logger.handlers
        if isinstance(h, logging.FileHandler)
    )

    file_handler.setLevel(logging.INFO)

    logging.info(message)

    file_handler.setLevel(logging.ERROR)

log_completion("Building finished.")

2. L'échelle de Température (Feedback)

Idéal pour des tableaux de bord de monitoring :

    0-9 : État "Froid" / Inactif (Gris/Bleu).

    10-19 : État "Normal" / Nominal (Vert/Cyan).

    20-29 : État "Chaud" / Attention (Orange).

    30-39 : État "Critique" / Danger (Rouge vif).

3. L'échelle Temporelle (Historique)

Pour représenter l'évolution d'une donnée :

    0-9 : Passé lointain (couleurs désaturées).

    10-19 : Tendance récente.

    20-29 : Valeur actuelle (couleur vive).

    30-39 : Prédictions / Futur (couleurs translucides ou pointillés).

.element {

@function map-reverse($map) {
  $keys: map-keys($map);
  $reversed-map: ();

  @for $i from length($keys) through 1 {
    $key: nth($keys, $i);
    $reversed-map: map-merge($reversed-map, ($key: map-get($map, $key)));
  }

  return $reversed-map;
}

$accents-reversed: map-reverse($accents);


@function map-shift($map, $steps: 1) {
  $keys: map-keys($map);
  $len: length($keys);
  $shifted-map: ();

  @for $i from 1 through $len {
    // Calcul du nouvel index circulaire
    $new-index: ($i + $steps - 1) % $len + 1;

    $old-key: nth($keys, $i);
    $new-val: map-get($map, nth($keys, $new-index));

    $shifted-map: map-merge($shifted-map, ($old-key: $new-val));
  }

  return $shifted-map;
}

$accents-shifted: map-shift($accents, 1);


@function map-pick($map, $keys-to-keep) {
  $filtered-map: ();

  @each $key in $keys-to-keep {
    @if map-has-key($map, $key) {
      $filtered-map: map-merge($filtered-map, ($key: map-get($map, $key)));
    }
  }

  return $filtered-map;
}

// Utilisation : on ne garde que la 1, la 4 et la 10
$accents-selective: map-pick($accents, 1 4 10);



@function transform-palette-ordered($map, $extract: null, $shift: 0, $reverse: false) {
  $result: $map;

  // 1. EXTRACTION (Pick)
  @if $extract != null {
    $result: map-pick($result, $extract);
  }

  // 2. SHIFT
  @if $shift != 0 {
    $result: map-shift($result, $shift);
  }

  // 3. REVERSE
  @if $reverse == true {
    $result: map-reverse($result);
  }

  return $result;
}

$theme-configs: (
  "default": (),

  // Exemple : on extrait 1, 2, 3, on décale de 1, puis on inverse
  "complex-variant": (
    extract: (1, 2, 3, 4, 5),
    shift: 2,
    reverse: true
  ),

  // Autre exemple : seulement un shift et un reverse
  "alt-theme": (
    shift: 1,
    reverse: true
  )
);

@each $name, $params in $theme-configs {
  [data-theme="#{$name}"] {
    // On récupère les valeurs avec des valeurs par défaut
    $extract: if(map-has-key($params, extract), map-get($params, extract), null);
    $shift:   if(map-has-key($params, shift), map-get($params, shift), 0);
    $reverse: if(map-has-key($params, reverse), map-get($params, reverse), false);

    $current-palette: transform-palette-ordered($accents, $extract, $shift, $reverse);

    @each $id, $color in $current-palette {
      --palAccent-#{$id}: #{$color};
    }
  }
}
