1. Le cas "Colorblind" (Accessibilité)

C'est l'exemple le plus courant. Une palette peut être à la fois Sequential (pour un dégradé) et Colorblind (parce que les contrastes de luminosité permettent aux personnes daltoniennes de distinguer les étapes).

    Exemple : La célèbre palette Viridis est techniquement Sequential, mais elle est presque toujours étiquetée Colorblind car elle est conçue pour être lisible par tous.

2. Le cas "Divergent" vs "Sequential"

Une palette Divergent est techniquement l'assemblage de deux palettes Sequential qui se rejoignent sur un point neutre (souvent le blanc ou le gris).

    Si vous n'utilisez qu'une moitié de la palette (du neutre vers un extrême), elle devient Sequential.

3. Le cas "Qualitative" vs "Sequential"

Certaines palettes Qualitative (couleurs distinctes) peuvent être perçues comme Sequential si les couleurs suivent une logique de luminosité (du plus clair au plus foncé), même si les teintes changent.

import sqlite3

def report_duplicates(db_path):
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()

    # Requête pour regrouper par hash et lister les noms
    query = '''
    SELECT
        hash_normal,
        COUNT(*) as nb,
        GROUP_CONCAT(stdname || ' (' || projname || ')', ' / ')
    FROM palettes
    GROUP BY hash_normal
    HAVING nb > 1
    ORDER BY nb DESC
    '''

    cursor.execute(query)
    rows = cursor.fetchall()

    if not rows:
        print("✅ Aucune palette en double trouvée.")
    else:
        print(f"🔍 {len(rows)} groupes de doublons détectés :\n")
        for hash_val, count, names in rows:
            print(f"[{count}x] Hash: {hash_val[:10]}...")
            print(f"    Palettes: {names}\n")

    conn.close()

def report_mirrors(db_path):
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()

    # On cherche où le normal de l'un égale le reverse de l'autre
    query = '''
    SELECT
        p1.stdname, p1.projname,
        p2.stdname, p2.projname
    FROM palettes p1
    JOIN palettes p2 ON p1.hash_normal = p2.hash_reverse
    WHERE p1.id < p2.id  -- Évite de trouver le même couple deux fois
    '''

    cursor.execute(query)
    rows = cursor.fetchall()

    for name1, proj1, name2, proj2 in rows:
        print(f"🔄 Miroir trouvé :")
        print(f"    A: {name1} [{proj1}]")
        print(f"    B: {name2} [{proj2}] (Inversée)\n")

    conn.close()


def report_name_conflicts(db_path):
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()

    # On cherche les lignes qui ont le même nom standard
    # mais des hashs différents
    query = '''
    SELECT
        p1.stdname,
        p1.projname, p1.hash_normal,
        p2.projname, p2.hash_normal
    FROM palettes p1
    JOIN palettes p2 ON p1.stdname = p2.stdname
    WHERE p1.id < p2.id
      AND p1.hash_normal != p2.hash_normal
      AND p1.hash_normal != p2.hash_reverse
    '''

    cursor.execute(query)
    rows = cursor.fetchall()

    if not rows:
        print("✅ Aucun conflit de nom trouvé (noms identiques = palettes identiques).")
    else:
        print(f"⚠️ Alerte : {len(rows)} conflits de contenu pour des noms identiques :\n")
        for name, proj1, hash1, proj2, hash2 in rows:
            print(f"Conflict sur '{name}':")
            print(f"  - Source A [{proj1}]: {hash1[:8]}...")
            print(f"  - Source B [{proj2}]: {hash2[:8]}...")
            print("-" * 30)

    conn.close()

@function map-reverse($map) {
  $keys: map-keys($map);
  $reversed-map: ();

  @for $i from length($keys) through 1 {
    $key: nth($keys, $i);
    $reversed-map: map-merge($reversed-map, ($key: map-get($map, $key)));
  }

  return $reversed-map;
}

$accents-reversed: map-reverse($accents);


@function map-shift($map, $steps: 1) {
  $keys: map-keys($map);
  $len: length($keys);
  $shifted-map: ();

  @for $i from 1 through $len {
    // Calcul du nouvel index circulaire
    $new-index: ($i + $steps - 1) % $len + 1;

    $old-key: nth($keys, $i);
    $new-val: map-get($map, nth($keys, $new-index));

    $shifted-map: map-merge($shifted-map, ($old-key: $new-val));
  }

  return $shifted-map;
}

$accents-shifted: map-shift($accents, 1);


@function map-pick($map, $keys-to-keep) {
  $filtered-map: ();

  @each $key in $keys-to-keep {
    @if map-has-key($map, $key) {
      $filtered-map: map-merge($filtered-map, ($key: map-get($map, $key)));
    }
  }

  return $filtered-map;
}

// Utilisation : on ne garde que la 1, la 4 et la 10
$accents-selective: map-pick($accents, 1 4 10);



@function transform-palette-ordered($map, $extract: null, $shift: 0, $reverse: false) {
  $result: $map;

  // 1. EXTRACTION (Pick)
  @if $extract != null {
    $result: map-pick($result, $extract);
  }

  // 2. SHIFT
  @if $shift != 0 {
    $result: map-shift($result, $shift);
  }

  // 3. REVERSE
  @if $reverse == true {
    $result: map-reverse($result);
  }

  return $result;
}

  // Autre exemple : seulement un shift et un reverse
