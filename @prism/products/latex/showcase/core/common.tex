\edef\nbcol{\palSize{\PALETTE}}


\begin{luacode*}
function generate_pattern(palname)
    local size = tonumber(token.get_macro("nbcol"))
    local palname = token.get_macro("PALETTE")

    local width = 16.0
    local height = 8.0
    local splitX = 4.0
    local jitter = 0.5
    local seed = 42

    local function getNextRandom()
        seed = (seed * 1664525 + 1013904223) % 4294967296
        
        return seed / 4294967296
    end

    local indices = {}

    for i = 1, size do 
        indices[i] = i
    end
    
    for i = size, 2, -1 do
        local j = math.floor(getNextRandom() * i) + 1

        indices[i], indices[j] = indices[j], indices[i]
    end

    local cols = math.ceil(math.sqrt(size / 2))
    local rows = math.ceil(size / (cols * 2))
    local cellW = (width - splitX) / cols
    local cellH = height / rows

    local points = {}
    
    for r = 0, rows do
        points[r] = {}
    
        for c = 0, cols do
            local x = splitX + (c * cellW)
            local y = r * cellH
    
            if c > 0 and c < cols then 
                x = x + (getNextRandom() - 0.5) * cellW * jitter
            end
            
            if r > 0 and r < rows then 
                y = y + (getNextRandom() - 0.5) * cellH * jitter 
            end

            points[r][c] = {x = x, y = y}
        end
    end

    local colorCounter = 1
    for r = 0, rows - 1 do
        for c = 0, cols - 1 do
            local p1, p2 = points[r][c], points[r][c+1]
            local p3, p4 = points[r+1][c], points[r+1][c+1]

            local triangles = {{p1, p2, p3}, {p2, p4, p3}}

            for _, t in ipairs(triangles) do
                local pIdx = indices[((colorCounter - 1) % size) + 1]
                local drawCmd = string.format(
                    "\\fill[\\palUse{%s}{%d}, draw=\\palUse{%s}{%d}] (%.3f,%.3f) -- (%.3f,%.3f) -- (%.3f,%.3f) -- cycle;",
                    palname,
                    pIdx,
                    palname,
                    pIdx, 
                    t[1].x, t[1].y, 
                    t[2].x, t[2].y,
                    t[3].x, t[3].y
                )
                
                tex.sprint(drawCmd)
                
                colorCounter = colorCounter + 1
            end
        end
    end
end
\end{luacode*}
