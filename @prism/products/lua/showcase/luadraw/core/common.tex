\directlua{
  PALSIZE = \string#PALETTE

  PALDIM   = 1
  PALDELTA = .3

  WIDTH    = 16
  NB_CELLS = 512

  RMAX = 8


  isintable = function(value, t)
    for _, v in ipairs(t) do
        if v == value then
          return true
        end
    end

    return false
  end


  drawPal = function(
    graphview,
    frame_color,
    draw_cell_border,
    pal,
    A,
    height,
    WIDTH,
    nbcells
  )
    local dl = WIDTH / nbcells

    for k = 1, nbcells do
      local color = palette(
        pal,
        (k - 1) / (nbcells - 1)
      )

      if draw_cell_border == true then
        tikz_options = "color=" .. frame_color
      else
        tikz_options = "color=" .. color
      end

      graphview:Drectangle(
        A,
        A + height,
        A + height + dl,
        tikz_options .. ", fill=" .. color
      )

      A = A + dl
    end

    graphview:Drectangle(
      A,
      A + height,
      A + height -WIDTH,
      "color=" .. frame_color
    )
  end
}

  )

% -- WAVES (small size) -- %

\begin{luadraw}{name = showcase-waves-\PALETTE}
if PALSIZE < 41 then
  yshift    = 0.25
  amplitude = 0.75

  xmax = WIDTH / 2
  xmin = -xmax

  ymax = amplitude + yshift*PALSIZE
  ymin = -ymax

  local g = graph:new{
    window = {
      xmin, xmax,
      ymin, ymax
    },
    bbox = false,
    bg   = COL_2
  }

  local ybase = ymax - amplitude

  for k = 1, PALSIZE do
    local color = palette(
      PALETTE,
      (k - 1) / (PALSIZE - 1)
    )

    local f = function(x)
      return ybase + amplitude*math.cos(1.05*x + k)
    end

    g:Dcartesian(
      f,
      {
        x            = {xmin, xmax},
        draw_options = "line width=2.5pt, color=" .. color
      }
    )

    ybase = ybase - yshift
  end

  g:Show()
end
\end{luadraw}

% -- RANDOM TRIANGLES (big size) -- %

\begin{luadraw}{name = showcase-random-triangles-\PALETTE}
if PALSIZE > 40 then
  xmax = 8
  xmin = -xmax

  ymax = xmax
  ymin = -ymax

  local g = graph:new{
    window = {
      xmin, xmax,
      ymin, ymax,
      .5, .5
    },
    bbox = false,
    bg   = COL_2
  }

  g:Linewidth(6)

  local nmax = math.floor(PALSIZE / 3)

  local pointsParCote = math.floor(
    (0.1 * PALSIZE) / 4
  )

  if pointsParCote < 1 then
    pointsParCote = 1
  end

  math.randomseed(20260208)

  local points = {
    Z(xmin, ymax),
    Z(xmin, ymin),
    Z(xmax, ymax),
    Z(xmax, ymin),
  }

  for cote = 1, 4 do
    for i = 1, pointsParCote do
      local x, y

      if cote == 1 then
        x, y = math.random(xmin, xmax), ymax

      elseif cote == 2 then
        x, y = math.random(xmin, xmax), ymin

      elseif cote == 3 then
        x, y = xmin, math.random(ymin, ymax)
      else
        x, y = xmax, math.random(ymin, ymax)
      end

      p = Z(x, y)

      if isintable(p, points) == false then
        table.insert(points, p)
      end
    end
  end

  local pointsRestants = nmax - #points - 4

  for i = 1, pointsRestants do
    p = Z(
      math.random(xmin + 1, xmax - 1),
      math.random(ymin + 1, ymax - 1)
    )

    if isintable(p, points) == false then
      table.insert(points, p)
    end
  end

  local T = delaunay(points)

  for k = 1, #T do
    g:Dpolyline(
      T[k],
      true,
         "color = " .. COL_2
      .. ", fill = " .. rgb(PALETTE[math.random(PALSIZE)])
    )
  end

  g:Show()
end
\end{luadraw}
