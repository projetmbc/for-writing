  local color = rgb(PALETTE[k])

    "color = black, fill = " .. color
  )

  A = A + h + delta
end

g:Show()

\begin{luadraw}{name = showcase-contour-\PALETTE}
local g = graph:new{
  window = {-1, 6.5, -1.5, 11},
  size   = {17, 15, 0},
  margin = {0, 0, 0, 0},
  bbox   = false
}

local i = cpx.I

local sin, cos = math.sin, math.cos

local f = function(x, y)
  return (x + y) / (2 + cos(x)*sin(y))
end

local Lz = range(1, 10)

local colors = getpalette(PALETTE, 10)

g:Dgradbox(
  {0, 5 + 10*i, 1, 1},
  {grid = true}
)

g:Linewidth(20)

g:Dcontour(
  f,
  Lz,
  {
    view   = {0, 5, 0, 10},
    colors = colors
  }
)

for k = 1, 10 do
  local y = (2*k + 4) / 3*i

  g:Dseg(
    {5.25 + y, 5.5 + y},
    1,
    "color = " .. colors[k]
  )

  g:Labelcolor(colors[k])

  g:Dlabel(
    "$z = " .. k .. "$",
    5.5 + y,
    {pos = "E"}
  )
end

g:Show()
\end{luadraw}

\begin{luadraw}{name = showcase-level-surface-\PALETTE}
local cos, sin = math.cos, math.sin, math.pi

local g = graph3d:new{
  window3d = {0, 5, 0, 10, 0, 11},
  adjust2d = true,
  size     = {17, 15, 0},
  bbox     = false,
  viewdir  = {220, 60},
  margin   = {0, 0, 0, 0}
}

g:Linewidth(2)

local S = surface(
  function(u, v)
    return M(u, v, (u + v) / (2 + cos(u)*sin(v)))
  end,
  0, 5, 0, 10,
  {30, 30}
)

local n = 10

local colors = getpalette(PALETTE, n, true)

local niv, S1 = {}

for k = 1, n do
  S1, S = cutfacet(S, {M(0, 0, k), -vecK})

  insert(
    niv,
    {
      S1,
      {
        color     = colors[k],
        mode      = mShaded,
        edgewidth = 0.5
      }
    }
  )
end

insert(
  niv,
  {
    S,
    {color = colors[n + 1]}
  }
)

g:Dboxaxes3d({
  grid      = true,
  gridcolor = "gray",
  fillcolor = "lightgray"
})

g:Dmixfacet(table.unpack(niv))

for k = 1, n do
  g:Dballdots3d(
    M(5, 0, k),
    rgb(colors[k])
  )
end

g:Show()
\end{luadraw}
